Inheritance

Last updated Nov 26, 2022

  • Oop
  • Nibm
  • Inheritance gives the ability to re-use attributes and methods of a class in another class
  • In simple terms inheritance is reuse
  • In inheritance there are two class
    • Super Class
    • Sub Class
  • The Super Class and the Sub Class has an “Is a” a relationship between them
  • The association “has a” relation between them
  • There are multiple types of inheritance
    • Single Inheritance
    • Hierarchical Inheritance
    • Multilevel Inheritance

# Single Inheritance

  • A class inherits another class is called single inheritance
  • In the top example Manger class inherits the Employee class
  • The inheritance done only in a one level Super class to Sub class

class Employee {
	protected String id;
	protected String name;
	protected float salary;

	public void setEmployeeData(String id, String  name, float salary) {
		this.id = id;
		this.name = name;
		this.salary = salary;
	}

	public void showEmployeeData() {
		System.out.println("Employee id - " + id);
		System.out.println("Employee name - " + name);
		System.out.println("Employee salary - " + salary);
	}
}

class Manager extends Employee {
	private String manager_id;
	private float incentive;
	private float salary_with_incentive;

	public void setMangerData(String manager_id, float incentive) {
		this.manager_id = manager_id;
		this.incentive = incentive;
	} 

	public void calcSalaryWithIncentive() {
		salary_with_incentive = salary + incentive;
	}

	public void showMangerData() {
		System.out.println("Employee id - " + id);
		System.out.println("Manger id - " + manager_id);
		System.out.println("Manger name - " + name);
		System.out.println("Manger salary with incentive - " + salary_with_incentive);
	}
} 

class Main {
	public static void main(String args[]) {
		Manager manager = new Manager();
		manager.setEmployeeData("E1", "Arkam", 100000);
		manager.setMangerData("M1", 50000);
		manager.calcSalaryWithIncentive();
		manager.showMangerData();
	}
}

class Employee {
	protected String id;
	protected String name;

	public void setEmployeeData(String id, String name) {
		this.id = id;
		this.name = name;
	}

	public void showEmployeeData() {
		System.out.println("Employee id - " + id);
		System.out.println("Employee name - " + name);
		System.out.print("\n");
	}
}

class Developer extends Employee {
	private String developer_id;

	public void setDeveloperData(String developer_id) {
		this.developer_id = developer_id;
	}

	public void showDeveloperData() {
		System.out.println("-- Role Developer --");
		System.out.println("Developer id - " + developer_id);
	}
}

class Manager extends Employee {
	private String manager_id;

	public void setManagerData(String manager_id) {
		this.manager_id = manager_id;
	}

	public void showManagerData() {
		System.out.println("-- Role Manager --");
		System.out.println("Manager id - " + manager_id);
	}
}

class Tester extends Employee {
	private String tester_id;

	public void setTesterData(String tester_id) {
		this.tester_id = tester_id;
	}

	public void showTesterData() {
		System.out.println("-- Role Tester --");
		System.out.println("Tester id - " + tester_id);
	}
}

class Main {
	public static void main(String args[]) {
		Developer developer = new Developer();
		developer.setEmployeeData("E1", "Gihane");
		developer.setDeveloperData("D1");
		developer.showDeveloperData();
		developer.showEmployeeData();

		Manager manager = new Manager();
		manager.setEmployeeData("E2", "Cadnus");
		manager.setManagerData("M1");
		manager.showManagerData();
		manager.showEmployeeData();

		Tester tester = new Tester();
		tester.setEmployeeData("E3", "Jane");
		tester.setTesterData("T1");
		tester.showTesterData();
		tester.showEmployeeData();
	}
}

class Employee {

	protected float salary;

		this.id = id;
		this.name = name;
		this.salary = salary;

class Manager extends Employee {

class AreaManager extends Manager {

}

# Inherited Constructor

  • When a constructor is derived the super class constructor will be called first then the derived class (sub class) constructor will be called. This happens because the constructor call happens from top to bottom
  • If there was a class our parent class is extending then the body of that class will be executed thereafter landing up to derived class
  • If we call a parameterized constructor of the base class then we call it using super()
  • The point to note is the base class constructor call must be the first line in the derived class constructor

class Employee {
	protected String id;
	protected String name;
	
	public Employee(String id, String name) {
		this.id = id;
		this.name = name;
	}
}

class Manager extends Employee {
	private String managing_sector;

	public Manager(String id, String name, String managing_sector) {
		// super is used to call the super class constructor 
		super(id, name);
		this.managing_sector = managing_sector;
	} 

	public void showManagerDetails() {
		System.out.println("Manager Id - " + id);
		System.out.println("Manager Name - " + name);
		System.out.println("Managing Sector  - " + id);
	}
}

class Main {
	public static void main(String args[]) {
		Manager manager = new Manager("E1", "Arkam", "DEV");
		manager.showManagerDetails();
	}
}
